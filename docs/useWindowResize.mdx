# useTouch

```js
import { useWindowResize } from 'use-events';
```

```jsx
const Example = () => {
  const [height, width] = useWindowResize();

  return (
    <div>
      Currently window size : { 'height is : ' + height } and { 'width is : ' + width }
    </div>
  );
};
```
